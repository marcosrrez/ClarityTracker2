# Integrated Session Intelligence Enhancement - Replit Agent Prompt

## Project Overview
Enhance existing Azure Speech-enabled behavioral health application by integrating Eleos-style features with advanced video analysis capabilities. Build comprehensive session intelligence that combines audio transcription, video analysis, and clinical AI into existing backend infrastructure.

    // Correlate audio sentiment with visual emotions
    const emotionalCorrelation = correlateEmotionalData(audio.emotions, video.emotions);
    
    // Detect incongruence between verbal and nonverbal
    const incongruenceAnalysis = detectIncongruence(
      transcript.sentiment,
      video.emotions,
      audio.tone
    );
    
    // Comprehensive engagement analysis
    const engagementAnalysis = fuseEngagementData(
      audio.speakingPatterns,
      video.engagementScore,
      transcript.participationMetrics
    );
    
    // Clinical theme correlation
    const clinicalCorrelation = correlateClinicalThemes(
      transcript.clinicalThemes,
      video.behavioralPatterns,
      audio.riskIndicators
    );
    
    // Treatment effectiveness prediction
    const treatmentPrediction = predictTreatmentEffectiveness({
      verbalContent: transcript,
      nonverbalBehavior: video,
      audioMetrics: audio,
      historicalData: await getHistoricalData(sessionId)
    });

    return {
      emotionalCorrelation,
      incongruenceAnalysis,
      engagementAnalysis,
      clinicalCorrelation,
      treatmentPrediction,
      overallInsights: generateOverallInsights({
        audio, video, transcript
      })
    };
  };

  // Generate clinical recommendations
  const generateClinicalRecommendations = (fusedInsights) => {
    const recommendations = [];

    // Based on incongruence analysis
    if (fusedInsights.incongruenceAnalysis.hasIncongruence) {
      recommendations.push({
        type: 'intervention',
        priority: 'high',
        title: 'Emotional Incongruence Detected',
        description: 'Patient verbal expression differs from nonverbal cues',
        suggestion: 'Consider exploring underlying emotions or barriers to expression',
        evidence: fusedInsights.incongruenceAnalysis.details
      });
    }

    // Based on engagement patterns
    if (fusedInsights.engagementAnalysis.averageEngagement < 0.6) {
      recommendations.push({
        type: 'engagement',
        priority: 'medium',
        title: 'Low Engagement Detected',
        description: 'Patient engagement below therapeutic threshold',
        suggestion: 'Consider adjusting therapeutic approach or addressing barriers',
        evidence: fusedInsights.engagementAnalysis.patterns
      });
    }

    // Based on risk indicators
    if (fusedInsights.clinicalCorrelation.riskLevel > 3) {
      recommendations.push({
        type: 'risk',
        priority: 'high',
        title: 'Elevated Risk Indicators',
        description: 'Multiple risk factors detected across modalities',
        suggestion: 'Conduct comprehensive risk assessment and safety planning',
        evidence: fusedInsights.clinicalCorrelation.riskFactors
      });
    }

    return recommendations;
  };

                  <span key={emotion.name} className="emotion-chip">
                    {emotion.name} ({emotion.confidence}%)
                  </span>
                ))}
              </div>
            </div>
          </div>

          <div className="overview-card engagement-analysis">
